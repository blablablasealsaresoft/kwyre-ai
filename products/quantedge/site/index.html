price_option.sh
# Price a call option with Black-Scholes
curl -X POST https://api.quantedge.io/v1/price/options \
  -H "Content-Type: application/json" \
  -d '{
    "spot": 100,
    "strike": 105,
    "rate": 0.05,
    "time_to_expiry": 0.25,
    "volatility": 0.2,
    "option_type": "call"
  }'

# Response:

  "price": 2.4634,
  "delta": 0.3842,
  "gamma": 0.0354,

  "vega": 0.1769,
  "rho": 0.0873
}
